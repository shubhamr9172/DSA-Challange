//User function Template for Java

class Compute {
    
    public void Rearrange(long a[], long n)
    {
        ArrayList<Long> al = new ArrayList<Long>();
       for(int i=0;i<n;i++)
       {
           if(a[i]<0)
           {
               al.add(a[i]);
           }
       }
       
       for(int i=0;i<n;i++)
       {
           if(a[i]>=0)
           {
               al.add(a[i]);
           }
       }
       
         for(int i=0;i<n;i++)
       {
         a[i]=al.get(i);
       }
       
    }
}
